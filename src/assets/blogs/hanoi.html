(function(){
    // 设计思想是 : 俩个类, 一个main函数
  // Rect 类封装了矩形(也就是一层楼)的所有属性及方法
  function Rect(){
      this.tower = null; // 这层楼现在属于那一座塔
  }
  Rect.prototype = {
      // 1.修改this.tower属性
        // 2.调用this.moveTo方法
        // 3.完成一层楼从一座塔移动到另一座塔
       moveTo : function () {
      return this.tower.pushRect(this, fn);
    }
  };
    
  // Tower类封装了塔的所有属性和方法
  function Tower(){
      this.rects = [];    // 该塔拥有的楼层
    // ...
  }
  Tower.prototype = {
    pushRect : function (rect, fn) {
        // 这是Tower类最核心的方法, 负责把一层楼移动到该座塔
        //  同时对dom的操作都在这里
      // rect 是Rect类的实例(对象)
      // fn 是回调函数, 因为在移动一层完成后才能移动另一层楼
    }
  };
  // 主函数
  function main(){}
    
  $(dom).on(click, function () {
      main();
  });
  /*
      PPS : 其实我不敢说这种设计的理念就是最好的, 但至少它的逻辑很好理解, 面向对象程序设计是一段很长的路, Keep Moving!
  */
  
})();
